pub type ExtensionCapabilities = BTreeMap<String, JsonObject>;
Expand description

MCP extension capabilities map.

Keys are extension identifiers in the format {vendor-prefix}/{extension-name} (e.g., io.modelcontextprotocol/ui, io.modelcontextprotocol/oauth-client-credentials). Values are per-extension settings objects. An empty object indicates support with no settings.

§Example

use rmcp::model::ExtensionCapabilities;
use serde_json::json;

let mut extensions = ExtensionCapabilities::new();
extensions.insert(
    "io.modelcontextprotocol/ui".to_string(),
    serde_json::from_value(json!({
        "mimeTypes": ["text/html;profile=mcp-app"]
    })).unwrap()
);
