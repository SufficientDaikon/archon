8/// Json wrapper for structured output
9///
10/// When used with tools, this wrapper indicates that the value should be
11/// serialized as structured JSON content with an associated schema.
12/// The framework will place the JSON in the `structured_content` field
13/// of the tool result rather than the regular `content` field.
14#[expect(clippy::exhaustive_structs, reason = "intentionally exhaustive")]
15pub struct Json<T>(pub T);
16
17// Implement JsonSchema for Json<T> to delegate to T's schema
18impl<T: JsonSchema> JsonSchema for Json<T> {
19    fn schema_name() -> Cow<'static, str> {
20        T::schema_name()
21    }
22
23    fn json_schema(generator: &mut schemars::SchemaGenerator) -> schemars::Schema {
24        T::json_schema(generator)
25    }
26}
27
28// Implementation for Json<T> to create structured content
29impl<T: Serialize + JsonSchema + 'static> IntoCallToolResult for Json<T> {
30    fn into_call_tool_result(self) -> Result<CallToolResult, crate::ErrorData> {
31        let value = serde_json::to_value(self.0).map_err(|e| {
32            crate::ErrorData::internal_error(
33                format!("Failed to serialize structured content: {}", e),
34                None,
35            )
36        })?;
37
38        Ok(CallToolResult::structured(value))
39    }
40}
