20/// This is an unified error type for the errors could be returned by the service.
21#[derive(Debug, thiserror::Error)]
22#[allow(clippy::large_enum_variant)]
23#[non_exhaustive]
24pub enum RmcpError {
25    #[cfg(any(feature = "client", feature = "server"))]
26    #[error("Service error: {0}")]
27    Service(#[from] crate::ServiceError),
28    #[cfg(feature = "client")]
29    #[error("Client initialization error: {0}")]
30    ClientInitialize(#[from] crate::service::ClientInitializeError),
31    #[cfg(feature = "server")]
32    #[error("Server initialization error: {0}")]
33    ServerInitialize(#[from] crate::service::ServerInitializeError),
34    #[error("Runtime error: {0}")]
35    Runtime(#[from] tokio::task::JoinError),
36    #[error("Transport creation error: {error}")]
37    // TODO: Maybe we can introduce something like `TryIntoTransport` to auto wrap transport type,
38    // but it could be an breaking change, so we could do it in the future.
39    TransportCreation {
40        into_transport_type_name: Cow<'static, str>,
41        into_transport_type_id: std::any::TypeId,
42        #[source]
43        error: Box<dyn std::error::Error + Send + Sync>,
44    },
45    // and cancellation shouldn't be an error?
46
47    // TODO: add more error variants as needed
48    #[error("Task error: {0}")]
49    TaskError(String),
50}
51
52impl RmcpError {
53    pub fn transport_creation<T: 'static>(
54        error: impl Into<Box<dyn std::error::Error + Send + Sync>>,
55    ) -> Self {
56        RmcpError::TransportCreation {
57            into_transport_type_id: std::any::TypeId::of::<T>(),
58            into_transport_type_name: std::any::type_name::<T>().into(),
59            error: error.into(),
60        }
61    }
